#include<stdio.h>
int main()
{
int intial_balance,deposit,balance_a,withdraw,balance_b,final_balance;
printf("intial_balance=");
scanf("%d",&intial_balance);
printf("deposit=");
scanf("%d",&deposit);
balance_a=intial_balance+deposit;
printf("balance_a=%d\n",balance_a);
printf("withdraw=");
scanf("%d",&withdraw);
balance_b=balance_a-withdraw;
printf("balance_b=%d\n",balance_b);
final_balance=balance_b;
printf("final_balance=%d",balance_b);
return 0;
}
