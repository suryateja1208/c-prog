#include<stdio.h>
int main()
{
   int i,factorial=1,number;
   printf("enter the number=");
   scanf("%d",&number);
   for(i=1;i<=number;i++)
   {
       factorial=factorial*i;
   }
   printf("factorial of %d =%d",number,factorial);
   return 0;
  }
